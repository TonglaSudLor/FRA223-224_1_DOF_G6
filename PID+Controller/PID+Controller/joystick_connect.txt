void HAL_UART_RxCpltCallback(UART_HandleTypeDef *huart) {
    if (huart->Instance == USART3) {

        // (ตัวเลือกเสริม) กระพริบไฟ LED ทันทีที่มีอะไรส่งเข้ามา เพื่อยืนยันว่าเข้า Interrupt แล้ว
        // HAL_GPIO_TogglePin(LD2_GPIO_Port, LD2_Pin);

        // ตรวจสอบว่าเป็นตัวจบข้อความ (Enter) หรือไม่
        if (rx_byte == '\n' || rx_byte == '\r') {
            if (rx_index > 0) {
                rx_buffer[rx_index] = '\0';          // ปิดท้าย String
                strcpy(controlState, rx_buffer);     // ก๊อปปี้ไปใช้งาน
                rx_index = 0;                        // รีเซ็ต index เพื่อรอรับคำสั่งใหม่

                // สลับสถานะไฟ LED เมื่อได้รับข้อความที่สมบูรณ์
                HAL_GPIO_TogglePin(LD2_GPIO_Port, LD2_Pin);
            }
        }
        else {
            // เก็บข้อมูลลง Buffer หากยังไม่เกินขนาด
            if (rx_index < sizeof(rx_buffer) - 1) {
                rx_buffer[rx_index++] = rx_byte;
            }
        }

        // เรียกใช้ฟังก์ชันรับค่าตัวต่อไป
        HAL_UART_Receive_IT(&huart3, &rx_byte, 1);
    }
}
